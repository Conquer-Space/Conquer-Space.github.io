138 Wallet& operator-=(const double amount) {
139 this->balance -= amount;
140 change -= amount;
141 GDP_change += amount;
142 // Record the money delta since last reset
143 return *this;
144 }
145
146 // Basic multiplication that logs the change
147 // TODO(EhWhoAmI): Make sure this is correct
148 Wallet& operator*=(const double coefficent) {
149 float newbalance = this->balance * coefficent;
150 float change = newbalance - this->balance;
151 if (change > 0) {
152 *this += change;
153 } else if (change < 0) {
154 *this -= change * -1;
155 }
156 return *this;
157 }
158
159 operator double() const { return balance; }
160
161 Wallet& operator=(double _balance) {
162 change += (_balance - balance);
163 if ((_balance - balance) < 0) {
164 GDP_change += _balance - balance;
165 }
166 balance = _balance;
167 return *this;
168 }
169
170 double GetBalance() const { return balance; }
171
172 void Reset() {
173 change = 0;
174 GDP_change = 0;

180 double change = 0;
