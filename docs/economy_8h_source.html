133 }

135 // TODO(EhWhoAmI): Make sure this is correct
136 Wallet& operator*=(const double coefficent) {
137 float newbalance = this->balance * coefficent;
138 float change = newbalance - this->balance;
139 if (change > 0) {
140 *this += change;
141 } else if (change < 0) {
142 *this -= change * -1;
143 }
144 return *this;
145 }
146
147 operator double() const { return balance; }
148
149 Wallet& operator=(double _balance) {
150 change += (_balance - balance);
151 if ((_balance - balance) < 0) {
152 GDP_change += _balance - balance;
153 }
154 balance = _balance;
155 return *this;
156 }
157
158 double GetBalance() const { return balance; }
159

161 change = 0;
162 GDP_change = 0;
163 }

168 double change = 0;
169 // Only records when spending money, so when money decreases
