118 void CalculatePeriod() {
119 T = 2 * PI * std::sqrt(semi_major_axis * semi_major_axis * semi_major_axis / Mu);

134inline glm::vec3 OrbitToVec3(const double& a, const double& e, const radian& i,
135 const radian& LAN, const radian& w, const radian& v) {
136 if (a == 0) {
137 return glm::vec3(0, 0, 0);
138 }
139 double r = (a * (1 - e * e)) / (1 + e * cos(v));
140 double x = r * cos(v);
141 double y = r * sin(v);
142 double z = 0;
143 glm::dvec3 o{x, y, z};
144
145 double rx = (o.x * (cos(w) * cos(LAN) - sin(w) * cos(i) * sin(LAN)) -
146 o.y * (sin(w) * cos(LAN) + cos(w) * cos(i) * sin(LAN)));
147 double ry = (o.x * (cos(w) * sin(LAN) + sin(w) * cos(i) * cos(LAN)) +
148 o.y * (cos(w) * cos(i) * cos(LAN) - sin(w) * sin(LAN)));
149 double rz = (o.x * (sin(w) * sin(i)) + o.y * (cos(w) * sin(i)));
150
151 // Convert to opengl coords
152 return glm::vec3{-rx, -rz, ry};
153}

161static double SolveKepler(const double& mean_anomaly, const double& ecc, const int steps = 200) {
162 // Because math is hard, we need to solve it iteratively
163 const int size = 10;
164 if (abs(ecc) < 1.0E-9) {
165 return mean_anomaly;
166 }
167 int maxit = steps;
168 int it = 0;
169
170 double de = 1000.0;
171
172 double ea = mean_anomaly;
173 double old_m = mean_anomaly;
174
175 while ((it < maxit) && (abs(de) > 1.0E-5)) { // normal accuracy is 1.0e-10
176 double new_m = ea - ecc * sin(ea);
177 de = (old_m - new_m) / (1.0 - ecc * cos(ea));
178 ea = ea + de;
179 it = it + 1;
180 }
181 return ea;
182}

187static radian TrueAnomaly(const Orbit& orbit, const second& time) {
188 // Calculate
189 double Mt = orbit.M0 + time * orbit.nu;
190 normalize_radian(Mt);
191 double E = SolveKepler(Mt, orbit.eccentricity);
192 return 2 * atan2 (sqrt(1 + orbit.eccentricity) * sin (E / 2), sqrt(1 - orbit.eccentricity) * cos (E / 2));
193}
