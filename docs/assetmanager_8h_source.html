74template <typename T>
75class ThreadsafeQueue {
76 std::queue<T> queue_;
77 mutable std::mutex mutex_;
78
79 // Moved out of public interface to prevent races between this
80 // and pop().
81 bool empty() const { return queue_.empty(); }
82
83 public:
84 ThreadsafeQueue() = default;
85 ThreadsafeQueue(const ThreadsafeQueue<T>&) = delete;
86 ThreadsafeQueue& operator=(const ThreadsafeQueue<T>&) = delete;
87
88 ThreadsafeQueue(ThreadsafeQueue<T>&& other) {
89 std::lock_guard<std::mutex> lock(mutex_);
90 queue_ = std::move(other.queue_);
91 }
92
93 virtual ~ThreadsafeQueue() {}
94
95 unsigned long size() const {
96 std::lock_guard<std::mutex> lock(mutex_);
97 return queue_.size();
98 }
99
100 std::optional<T> pop() {
101 std::lock_guard<std::mutex> lock(mutex_);
102 if (queue_.empty()) {
103 return {};
104 }
105 T tmp = queue_.front();
106 queue_.pop();
107 return tmp;
108 }
109
110 void push(const T& item) {
111 std::lock_guard<std::mutex> lock(mutex_);
112 queue_.push(item);
113 }
114};
