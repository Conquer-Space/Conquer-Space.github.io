79template <typename T>
80class ThreadsafeQueue {
81 std::queue<T> queue_;
82 mutable std::mutex mutex_;
83
84 // Moved out of public interface to prevent races between this
85 // and pop().
86 bool empty() const { return queue_.empty(); }
87
88 public:
89 ThreadsafeQueue() = default;
90 ThreadsafeQueue(const ThreadsafeQueue<T>&) = delete;
91 ThreadsafeQueue& operator=(const ThreadsafeQueue<T>&) = delete;
92
93 ThreadsafeQueue(ThreadsafeQueue<T>&& other) {
94 std::lock_guard<std::mutex> lock(mutex_);
95 queue_ = std::move(other.queue_);
96 }
97
98 virtual ~ThreadsafeQueue() {}
99
100 unsigned long size() const {
101 std::lock_guard<std::mutex> lock(mutex_);
102 return queue_.size();
103 }
104
105 std::optional<T> pop() {
106 std::lock_guard<std::mutex> lock(mutex_);
107 if (queue_.empty()) {
108 return {};
109 }
110 T tmp = queue_.front();
111 queue_.pop();
112 return tmp;
113 }
114
115 void push(const T& item) {
116 std::lock_guard<std::mutex> lock(mutex_);
117 queue_.push(item);
118 }
119};

184 static_assert(std::is_base_of<Asset, T>::value, "Class is not child of cqsp::asset::Asset");
185 std::size_t separation = key.find(":");
186 // Default name is core
187 std::string package_name = "core";
188 if (separation != std::string::npos) {
189 package_name = key.substr(0, separation);
190 }
191
192 // Check if package exists
193 if (packages.count(package_name) == 0) {
194 ENGINE_LOG_ERROR("Cannot find package {}", package_name);
195 }
196 std::string pkg_key = key.substr(separation + 1, key.length());
197 auto& package = packages[package_name];
198 // Probably a better way to do this, to be honest
