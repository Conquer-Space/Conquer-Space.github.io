80template <typename T>
81class ThreadsafeQueue {
82 std::queue<T> queue_;
83 mutable std::mutex mutex_;
84
85 // Moved out of public interface to prevent races between this
86 // and pop().
87 bool empty() const { return queue_.empty(); }
88
89 public:
90 ThreadsafeQueue() = default;
91 ThreadsafeQueue(const ThreadsafeQueue<T>&) = delete;
92 ThreadsafeQueue& operator=(const ThreadsafeQueue<T>&) = delete;
93
94 ThreadsafeQueue(ThreadsafeQueue<T>&& other) {
95 std::lock_guard<std::mutex> lock(mutex_);
96 queue_ = std::move(other.queue_);
97 }
98
99 virtual ~ThreadsafeQueue() {}
100
101 unsigned long size() const {
102 std::lock_guard<std::mutex> lock(mutex_);
103 return queue_.size();
104 }
105
106 std::optional<T> pop() {
107 std::lock_guard<std::mutex> lock(mutex_);
108 if (queue_.empty()) {
109 return {};
110 }
111 T tmp = queue_.front();
112 queue_.pop();
113 return tmp;
114 }
115
116 void push(const T& item) {
117 std::lock_guard<std::mutex> lock(mutex_);
118 queue_.push(item);
119 }
120};
